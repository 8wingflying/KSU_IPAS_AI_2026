## List
```
# 1. 列表總和
nums = [3, 5, 7, 2]
print(sum(nums))

# 2. 最大值與最小值
nums = [3, -2, 9, 5]
print("max =", max(nums), ", min =", min(nums))

# 3. 計算偶數個數
nums = [1, 2, 4, 7, 10]
count_even = sum(1 for x in nums if x % 2 == 0)
print(count_even)

# 4. 去除重複值（保留順序）
nums = [1, 2, 2, 3, 3, 3, 4]
result = []
for x in nums:
    if x not in result:
        result.append(x)
print(result)

# 5. 倒序列表（不用 reverse）
nums = [1, 2, 3, 4, 5]
rev = nums[::-1]
print(rev)

# 6. 找出指定元素的所有索引
nums = [1, 3, 7, 3, 2, 3]
target = 3
indices = [i for i, v in enumerate(nums) if v == target]
print(indices)

# 7. 列表平移（旋轉）
nums = [1, 2, 3, 4, 5]
k = 2
k = k % len(nums)
rotated = nums[-k:] + nums[:-k]
print(rotated)

# 8. 列表元素平方
nums = [1, 2, 3]
squared = [x ** 2 for x in nums]
print(squared)

# 9. 過濾奇數
nums = [1, 2, 3, 4, 5, 6]
odds = [x for x in nums if x % 2 == 1]
print(odds)

# 10. 子列表長度
lists = [[1, 2], [3, 4, 5], [6]]
lengths = [len(sub) for sub in lists]
print(lengths)

# 11. 合併兩個列表（交錯）
a = [1, 3, 5]
b = [2, 4, 6]
merged = []
for x, y in zip(a, b):
    merged.append(x)
    merged.append(y)
print(merged)

# 12. 判斷是否為遞增序列
nums = [1, 3, 5, 7]
is_inc = all(nums[i] < nums[i+1] for i in range(len(nums)-1))
print(is_inc)

# 13. 計算出現次數（頻率表）
items = ['a', 'b', 'a', 'c']
freq = {}
for v in items:
    freq[v] = freq.get(v, 0) + 1
print(freq)

# 14. 列表扁平化
nested = [[1, 2], [3, 4], [5]]
flat = [x for sub in nested for x in sub]
print(flat)

# 15. 找出第二大值（走訪版）
nums = [5, 9, 1, 7, 3]
first = second = float('-inf')
for x in nums:
    if x > first:
        second = first
        first = x
    elif first > x > second:
        second = x
print(second)

# 16. 移除列表中的所有 None
data = [1, None, 2, None, 3, 4, None]
cleaned = [x for x in data if x is not None]
print(cleaned)

# 17. 依長度排序字串列表
words = ["apple", "hi", "python", "a"]
sorted_words = sorted(words, key=lambda s: (len(s), s))
print(sorted_words)

# 18. 兩列表共同元素（交集）
a = [1, 2, 3, 4]
b = [3, 4, 4, 5, 6]
common = sorted(set(a) & set(b))
print(common)

# 19. 分組切片
nums = [1, 2, 3, 4, 5, 6, 7]
k = 3
groups = [nums[i:i+k] for i in range(0, len(nums), k)]
print(groups)

# 20. 簡易購物車
cart = []
while True:
    print("\n1) 加入商品  2) 列出商品  3) 刪除商品  4) 結束")
    choice = input("請選擇功能: ")

    if choice == "1":
        name = input("輸入商品名稱: ")
        cart.append(name)
        print("已加入", name)
    elif choice == "2":
        print("目前購物車:", cart)
    elif choice == "3":
        name = input("要刪除的商品名稱: ")
        if name in cart:
            cart.remove(name)
            print("已刪除", name)
        else:
            print("購物車中沒有此商品")
    elif choice == "4":
        print("程式結束")
        break
    else:
        print("輸入錯誤，請重新選擇")
```
